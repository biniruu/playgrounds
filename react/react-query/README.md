- @typescript-eslint/eslint-plugin
- @typescript-eslint/parser
- @types/jest
- @types/react-test-renderer
- eslint-config-prettier
- eslint-plugin-jest
- eslint-plugin-prettier
- postcss-cli
- postcss-html
- postcss-syntax
- postcss-url
- prettier
- prettier-plugin-tailwindcss
- stylelint
- stylelint-config-standard
- stylelint-order
- ts-jest
